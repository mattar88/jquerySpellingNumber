jQuery Spelling Number


Spelling in English

1. Example Convert Number into words

Code

    var options_sn = {
            lang: "en",
            digitsLengthW2F: 3,
            decimalSeperator: "point"
           };
           
     $.spellingNumber(1625444.13, options_sn)

Number and Selling value

1625444.13
one million six hundred twenty-five thousand four hundred and forty-four point one hundred thirty

2. Example Convert Dollars and Cents into words

Code

    var options_sn = {
        lang: "en",
        wholesUnit: "dollars",
        fractionUnit: "cents",
        digitsLengthW2F: 2,
        decimalSeperator: "and"
           };

    $.spellingNumber(12300.23, options_sn);

Number and Selling value

12300.23
twelve thousand three hundred dollars and twenty-three cents

3. Example Convert Weight number into words

Code

    var options_sn = {
        lang: "en",
        wholesUnit: "ton",
        fractionUnit: "kg",
        digitsLengthW2F: 3,
        decimalSeperator: "and"
           };
           
  $.spellingNumber(100.951, options_sn);

Number and Selling value

100.951
one hundred ton and nine hundred fifty-one kg

Code
